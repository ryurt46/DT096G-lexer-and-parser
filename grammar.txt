<match>    := <expr>
<expr>     := <subexpr> [<expr>]
<subexpr>  := <group> | <or> | <word> | <any>
<group>    := '(' <expr> ')'
<or>       := <word> '+' <word>
<word>     := <term> [<word>]
<term>     := <count> | <greedy> | <char>
<count>    := <char> '{' <digit> '}' | <any> '{' <digit> '}'
<greedy>   := <char> '*' | <any> '*'
<char>     := [a-zA-Z]
<digit>    := [0-9]
<any>      := '.'
