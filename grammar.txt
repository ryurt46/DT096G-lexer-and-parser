<match>    := <expr>
<expr>     := <subexpr>
<subexpr>  := <or> | <multiple> | <group> | <any> | <count> | <ignore> | <word>
<or>       := <expr> '+' <expr>
<multiple> := <expr> '*'
<group>    := '(' <expr> ')'
<any>      := '.'
<count>    := <expr> '{' <digit> '}'
<ignore>   := <expr> '\I'
<word>     := [char] [<word>]
<digit>   := [0-9]+
<char>     := [a-zA-Z]
